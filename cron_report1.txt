#!/usr/bin/perl
use strict; 
use warnings;

##### Generating filename ##################################
my($sec, $min, $hour, $mday,$mon,$year,$wday,$yday,$idst)=localtime(); 
my $hostname = qx/hostname/;
$hostname =~ s/\n//;
my $file = $hostname.$mday.$mon.$year.'.txt';

############### FILES #############

my @cron; 
my $filename = 'democron.txt';

open(my $fh, '<:encoding(UTF-8)', $filename)
  or die "Could not open file '$filename' $!";
 
while (my $row = <$fh>) {

  chomp $row;

  push @cron ,$row;   
}

foreach(@cron){

  my($con1,$cron2,$cron3,$cron4,$cron5,$script)  =split/ /,$_;

  print"$script\n"; 

}
